# 1 求集合数据的均方差
```c
#include "stdio.h"
#include "math.h"
int main()
{
    int N,i;
    scanf("%d",&N);
    double Avg,sum=0,A[N];
    for(i=0;i<N;i++){
        scanf("%lf",&A[i]);
        Avg+=A[i];
    }
    Avg/=N;
    for(i=0;i<N;i++){
        A[i]*=A[i]=A[i]-Avg;
        sum+=A[i];
    }
    sum=sqrt(sum/N);
    printf("%.5lf",sum);
}
```

# 2 组合数的和
```c
#include "stdio.h"
int main()
{
    int i,x,N;
    scanf("%d",&N);
    int sum=0,A[N],temp;
    for(int i=0;i<N;i++)	
        scanf("%d",&A[i]);
    for(i=0;i<N;i++){
        for(x=0;x<N;x++){
            if(i!=x){
                temp=A[i]*10+A[x];
                sum+=temp;
            }
        }
    }
    printf("%d",sum);
}
```
# 3 简化的插入排序
```c
#include "stdio.h"
int main()
{
    int N;
    scanf("%d",&N);
    int A[N],x;
    for(int i=0;i<N;i++)
        scanf("%d",&A[i]);
    scanf("%d",&x);
    if(N==0)
        printf("%d ",x);
    
    for(int i=0;i<N;i++){
        if(A[i]>x){
            if(x>A[i-1])
                printf("%d ",x);
        }	
        printf("%d ",A[i]);
        if(i==N-1&&x>A[i]&&N!=1)
            printf("%d ",x);
    }
}
```
