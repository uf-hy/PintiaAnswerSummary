# 7-2 比较大小
```c
#include "stdio.h"
int main()
{
	int a,b,c,i;
	scanf("%d%d%d",&a,&b,&c);
	for(i=0;i<2;i++)
	{
		if(a>=b)
		{
			a=a+b;
			b=a-b;
			a=a-b;
		}
		if(b>=c)
		{
			b=c+b;
			c=b-c;
			b=b-c;
		}
	}
	printf("%d->%d->%d",a,b,c);
}
```

# 7-3 谁是赢家

if(a1>=a2&&c1+c2+c3!=3)
{
	printf("The winner is a: %d + %d",a1,3-c1-c2-c3);
