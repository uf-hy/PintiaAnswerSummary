	deIete(p,n);
	printf("%s",a);
	return 0;
}
void deIete(char *p, int n)
{
	for(int i=n-1;*(p+i)!='\0';i++)
		*(p+i)=*(p+i+1);
}

{
	int n=strlen(p)-1,i=0;

		*(p+65535)=*(p+i);
		*(p+i)=*(p+n-i);
