# 1 求集合数据的均方差
```c
#include "stdio.h"
#include "math.h"
int main()
{
    int N,i;
    scanf("%d",&N);
    double Avg,sum=0,A[N];
    for(i=0;i<N;i++){
        scanf("%lf",&A[i]);
        Avg+=A[i];
    }
    Avg/=N;
    for(i=0;i<N;i++){
        A[i]*=A[i]=A[i]-Avg;
        sum+=A[i];
    }
    sum=sqrt(sum/N);
    printf("%.5lf",sum);
}
```

int main()
{

    scanf("%d",&N);
    int sum=0,A[N],temp;
    for(int i=0;i<N;i++)	
        scanf("%d",&A[i]);
    for(i=0;i<N;i++){
        for(x=0;x<N;x++){
            if(i!=x){
                temp=A[i]*10+A[x];

            }
        }
    }

# 3 简化的插入排序
```c
#include "stdio.h"
int main()
{
	int N;
	scanf("%d",&N);
	int A[N],x,IO=0;
	for(int i=0;i<N;i++)
		scanf("%d",&A[i]);
	scanf("%d",&x);
	if(N==0)
		printf("%d ",x);
	
	for(int i=0;i<N;i++){
		if(A[i]>x&&IO==0){
			printf("%d ",x);
			IO++;		
		}	
		printf("%d ",A[i]);
		if(i==N-1&&IO==0)
			printf("%d ",x);
	}
}
```
# 4 倒数第N个字符串
```c
#include "stdio.h"
#include "math.h"
int main()
{
	int N;
	scanf("%d",&N);
	int m=1,a1,a2,num[N];
	m=pow(26,N);
	scanf("%d",&a1);
	a2=m-a1;
	for(int t=N;t!=0;t--){
		num[t-1]=a2/((int)pow(26,t-1))%26;
	}
	for(;N!=0;N--){
		printf("%c",97+num[N-1]);
	}
}
```
