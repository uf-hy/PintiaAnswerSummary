# 1 求集合数据的均方差
```c
#include "stdio.h"
#include "math.h"
int main()
{
	int N,i;
	scanf("%d",&N);
	double Avg,sum=0,A[N];
	for(i=0;i<N;i++){
		scanf("%lf",&A[i]);
		Avg+=A[i];
	}
	Avg/=N;
	for(i=0;i<N;i++){
		A[i]*=A[i]=A[i]-Avg;
		sum+=A[i];
	}
	sum=sqrt(sum/N);
	printf("%.5lf",sum);
}
```
